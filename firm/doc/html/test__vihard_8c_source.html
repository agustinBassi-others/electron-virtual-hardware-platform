280 static void TestVhIntegral(){
281  const uint32_t timeToUpdateLcd = 35;
282  const uint32_t timeToUpdate7Segs = 8;
283  const uint32_t timeToToggleLed = 8;
284  uint32_t counterUpdateLcd = 0;
285  uint32_t counterUpdate7Segs = 0;
286  uint32_t counterToggleLed = 0;
287 
288  uint8_t value7Segs = '0';
289 
290  uint16_t adcValue = 0;
291 
292  bool_t stateTec1 = TRUE;
293  bool_t stateTec2 = TRUE;
294 
295  uint8_t lcdMessageIndex = 0;
296  char lcdMessages[4][50] = {
297  "Hola desde ViHard! Este es un mensaje multilinea",
298  "Mensaje linea 1",
299  "Mensaje linea 2",
300  "Mensaje linea 3"
301  };
302 
303 
304  while(1){
305  // Lee el ADC virtual y envia el valor al DAC virtual
306  adcValue = Vh_AdcRead(VH_ADC_CH1);
307  Vh_DacWrite(VH_DAC_CH1, adcValue);
308  // Enciende VH_LED1 si VHTEC1 esta pulsada, sino lo apaga
309  stateTec1 = Vh_GpioRead(VH_TEC1);
310  Vh_GpioWrite(VH_LED1, !stateTec1);
311  // Enciende VH_LED2 si VHTEC2 esta pulsada, sino lo apaga
312  stateTec2 = Vh_GpioRead(VH_TEC2);
313  Vh_GpioWrite(VH_LED2, !stateTec2);
314  // Pasado un tiempo togglea led virtual y led fisico de la EDU CIAA
315  if (++counterToggleLed >= timeToToggleLed){
316  // Toggle del pin virtual VH_LED3
317  Vh_GpioToggle(VH_LED4);
318  // Toggle del led fisico de la EDU CIAA
319  gpioToggle(LED3);
320  counterToggleLed = 0;
321  }
322  // Pasado un tiempo actualiza el valor del display 7 segmentos
323  if (++counterUpdate7Segs >= timeToUpdate7Segs){
324  Vh_7SegmentsWrite(VH_7SEG, value7Segs);
325  if (++value7Segs > '9'){
326  value7Segs = '0';
327  }
328  counterUpdate7Segs = 0;
329  }
330  // Pasado un tiempo actualiza el valor del display LCD
331  if (++counterUpdateLcd >= timeToUpdateLcd){
332  Vh_LcdWriteString(
333  VH_LCD1,
334  (LcdLine_t) (lcdMessageIndex + '0'),
335  lcdMessages[lcdMessageIndex]
336  );
337  if (++lcdMessageIndex > 3){
338  lcdMessageIndex = 0;
339  }
340  counterUpdateLcd = 0;
341  }
342  }
343 }
