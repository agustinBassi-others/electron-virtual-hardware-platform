196 static bool_t AnalogValueToString (uint16_t numberToConvert, char * strNumber)
197 {
198  bool_t error = FALSE;
199  uint8_t thousands = 0;
200  uint8_t hundreds = 0;
201  uint8_t tens = 0;
202  uint8_t units = 0;
203 
204  if (numberToConvert <= MAX_ANALOG_VALUE)
205  {
206  thousands = numberToConvert / 1000;
207  hundreds = numberToConvert / 100;
208  if (hundreds >= 10)
209  {
210  hundreds = 0;
211  }
212  tens = numberToConvert;
213  tens -= ((thousands * 1000) + (hundreds * 100));
214  tens /= 10;
215  units = numberToConvert;
216  units -= ((thousands * 1000) + (hundreds * 100) + (tens * 10));
217 
218  strNumber[0] = thousands + '0';
219  strNumber[1] = hundreds + '0';
220  strNumber[2] = tens + '0';
221  strNumber[3] = units + '0';
222  strNumber[4] = '\0';
223  }
224  else
225  {
226  error = TRUE;
227  }
228  return !error;
229 }

241 static bool_t CheckIfValidCommand (ViHardCommand_t command,
242  ViHardPeriph_t perMap)
243 {
244  bool_t isValidCommand = FALSE;
245  if (command == VH_GPIO_READ)
246  {
247  if (perMap == VH_TEC1 || perMap == VH_TEC2 ||
248  perMap == VH_TEC3 || perMap == VH_TEC4)
249  {
250  isValidCommand = TRUE;
251  }
252  if (perMap == VH_LEDR || perMap == VH_LEDG || perMap == VH_LEDB ||
253  perMap == VH_LED1 || perMap == VH_LED2 || perMap == VH_LED3 ||
254  perMap == VH_LED4)
255  {
256  isValidCommand = TRUE;
257  }
258  }
259  else if (command == VH_GPIO_WRITE)
260  {
261  if (perMap == VH_LEDR || perMap == VH_LEDG || perMap == VH_LEDB ||
262  perMap == VH_LED1 || perMap == VH_LED2 || perMap == VH_LED3 ||
263  perMap == VH_LED4)
264  {
265  isValidCommand = TRUE;
266  }
267  }
268  else if (command == VH_ADC_READ)
269  {
270  if (perMap == VH_ADC_CH1)
271  {
272  isValidCommand = TRUE;
273  }
274  }
275  else if (command == VH_DAC_WRITE)
276  {
277  if (perMap == VH_DAC_CH1)
278  {
279  isValidCommand = TRUE;
280  }
281  }
282  else if (command == VH_LCD_WRITE_BYTE || command == VH_LCD_WRITE_STRING)
283  {
284  if (perMap == VH_LCD1)
285  {
286  isValidCommand = TRUE;
287  }
288  }
289  else if (command == VH_7SEG_WRITE)
290  {
291  if (perMap == VH_7SEG)
292  {
293  isValidCommand = TRUE;
294  }
295  }
296 
297  if (isValidCommand)
298  {
299  DelayMs(MS_BETWEEN_COMMANDS);
300  }
301 
302  return isValidCommand;
303 }

355 bool_t Vh_GpioRead (ViHardPeriph_t gpioPin)
356 {
357  char stringCommand[10];
358  bool_t pinState = TRUE;
359  uint8_t dataSerial = 0;
360  uint16_t counter = 0;
361  uint8_t i = 0;
362  bool_t flagCommandInit = FALSE;
363 
364  if (CheckIfValidCommand(VH_GPIO_READ, gpioPin))
365  {
366  stringCommand[0] = COMMAND_INIT;
367  stringCommand[1] = VH_GPIO_READ;
368  stringCommand[2] = COMMAND_SEPARATOR;
369  stringCommand[3] = gpioPin;
370  stringCommand[4] = COMMAND_SEPARATOR;
371  stringCommand[5] = VH_COMM_REQUEST;
372  stringCommand[6] = COMMAND_END;
373  stringCommand[7] = '\n';
374  stringCommand[8] = '\0';
375 
376  UartWriteString(stringCommand);
377 
378  // limpia el buffer
379  for (i = 0; i < 10; i++)
380  {
381  stringCommand[i] = 0;
382  }
383  i = 0;
384 
385  // Espera a recibir data por un tiempo determinado
386  while (++counter < 1000 && i < 10)
387  {
388  if ((dataSerial = UartReadByte()) != 0)
389  {
390  if (dataSerial == COMMAND_INIT)
391  {
392  flagCommandInit = TRUE;
393  }
394  if (flagCommandInit)
395  {
396  stringCommand[i] = dataSerial;
397  if (stringCommand[i] == '}')
398  {
399  break;
400  }
401  i++;
402  }
403  }
404  }
405 
406  // chequea si salio por timeout
407  if (counter < 1000)
408  {
409  // chequea que todos lo que haya llegado
410  // sea una respuesta correcta
411  if (stringCommand[0] == COMMAND_INIT &&
412  stringCommand[1] == VH_GPIO_READ &&
413  stringCommand[2] == COMMAND_SEPARATOR &&
414  stringCommand[4] == COMMAND_SEPARATOR &&
415  stringCommand[5] == VH_COMM_RESPONSE &&
416  stringCommand[6] == COMMAND_SEPARATOR &&
417  (stringCommand[7] == VH_GPIO_LOW ||
418  stringCommand[7] == VH_GPIO_HIGH))
419  {
420  pinState = stringCommand[7] - '0';
421  }
422  }
423  }
424 
425  return pinState;
426 }
